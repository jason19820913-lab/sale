📋 Google Sheets 串接設定

結帳後系統將自動把每筆訂單傳送至 Google 試算表。請依步驟完成設定：

1
開啟您的 Google 試算表 → 「擴充功能」→「Apps Script」
2
複製下方程式碼，全部取代原有內容後儲存
3
「部署」→「新增部署作業」→ 類型：網頁應用程式，存取權限：所有人
4
複製部署後的 URL，貼入下方並儲存
📄 APPS SCRIPT 程式碼（動態欄位版）
🆕 動態欄位版：新增商品後不需重新部署，Sheet 會自動在右側新增欄位
// 擇木良食 × 金鶯診所｜銷售記錄接收器（動態欄位版） // 新增商品不需重新部署，doPost 會自動在 Sheet 新增欄位 const SHEET_NAME = '銷售紀錄'; // 固定欄位（前段 + 後段），商品欄位動態插入在中間 const FIXED_FRONT = ['日期','時間','院區','顧客姓名','身分證遮罩','病歷號','當班醫師','櫃檯人員']; const FIXED_BACK = ['結帳金額','付款方式','醫師抽成','員工抽成','藥師抽成','身分證雜湊']; function getOrCreateSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let s = ss.getSheetByName(SHEET_NAME); if (!s) { s = ss.insertSheet(SHEET_NAME); s.appendRow([...FIXED_FRONT, ...FIXED_BACK]); s.getRange('1:1').setFontWeight('bold').setBackground('#2C3E35').setFontColor('#fff'); s.setFrozenRows(1); } return s; } // 確保 Sheet 有這個商品欄，沒有就在 FIXED_BACK 前面插入 function ensureProductCol(sheet, colName) { const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; if (headers.includes(colName)) return; // 找到 FIXED_BACK 第一欄（結帳金額）的位置，插入在它之前 const insertBefore = headers.indexOf(FIXED_BACK[0]); const insertCol = (insertBefore >= 0 ? insertBefore : headers.length) + 1; sheet.insertColumnBefore(insertCol); sheet.getRange(1, insertCol).setValue(colName) .setFontWeight('bold').setBackground('#2C3E35').setFontColor('#fff'); } function doPost(e) { try { const d = JSON.parse(e.postData.contents); const qty = d.itemQty || {}; const sheet = getOrCreateSheet(); // 確保所有本次訂單的商品欄都存在 Object.keys(qty).forEach(col => ensureProductCol(sheet, col)); // 重新讀取最新欄位順序 const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; // 依欄位順序組出這筆資料 const fixedData = { '日期': d.date, '時間': d.time, '院區': d.clinic||'', '顧客姓名': d.customer, '身分證遮罩': d.idMask||'', '病歷號': d.phone||'', '當班醫師': d.doctor, '櫃檯人員': d.staff||'', '結帳金額': d.total, '付款方式': d.payMethod, '醫師抽成': d.doctorComm||0, '員工抽成': d.staffComm||0, '藥師抽成': d.pharmaComm||0, '身分證雜湊': d.idHash||'' }; const row = headers.map(h => qty[h] !== undefined ? qty[h] : (fixedData[h] !== undefined ? fixedData[h] : '')); sheet.appendRow(row); return json({status:'ok'}); } catch(err) { return json({status:'error',msg:err.message}); } } function doGet(e) { const p = (e||{}).parameter||{}; if (p.action==='query'&&p.idHash) { try { const sheet = getOrCreateSheet(); const rows = sheet.getDataRange().getValues(), hdrs = rows[0]; const HC = hdrs.indexOf('身分證雜湊'); const productCols = hdrs.filter(h => !FIXED_FRONT.includes(h) && !FIXED_BACK.includes(h)); const records = rows.slice(1).filter(r=>r[HC]===p.idHash).map(r=>{ const o={}; hdrs.forEach((h,i)=>o[h]=r[i]); o.items = productCols.filter(c=>o[c]>0).map(c=>`${c}×${o[c]}`).join('、'); o.date=o['日期']; o.time=o['時間']; o.clinic=o['院區']; o.customer=o['顧客姓名']; o.doctor=o['當班醫師']; o.staff=o['櫃檯人員']; o.total=o['結帳金額']; o.payMethod=o['付款方式']; return o; }); return json({status:'ok',records}); } catch(err){return json({status:'error',msg:err.message});} } return json({status:'ok',msg:'連線成功（動態欄位版）'}); } function json(o){return ContentService.createTextOutput(JSON.stringify(o)).setMimeType(ContentService.MimeType.JSON);}
