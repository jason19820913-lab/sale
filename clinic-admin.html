📋 Google Sheets 串接設定

結帳後系統將自動把每筆訂單傳送至 Google 試算表。請依步驟完成設定：

1
開啟您的 Google 試算表 → 「擴充功能」→「Apps Script」
2
複製下方程式碼，全部取代原有內容後儲存
3
「部署」→「新增部署作業」→ 類型：網頁應用程式，存取權限：所有人
4
複製部署後的 URL，貼入下方並儲存
📄 APPS SCRIPT 程式碼（通用版 — 只需部署一次）
✅ 通用版：GAS 不含任何商品/人員/院區邏輯，之後不論怎麼新增功能都不需要重新部署
// 擇木良食 × 金鶯診所｜通用版 GAS（只需部署一次，永遠不用改） // ───────────────────────────────────────────────────────────── // doGet?sheet=工作頁名稱 → 讀取整頁所有資料（JSON） // doPost {sheet, headers, rows} → 覆寫整頁（商品/人員/院區） // doPost {sheet, row, append:true} → 新增一列（銷售記錄） // ───────────────────────────────────────────────────────────── // 工作頁名稱（前後台 JS 定義，GAS 完全不需知道）： // 銷售紀錄 / 商品主檔 / 人員名單 / 院區清單 function getOrCreate(name, headers) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let s = ss.getSheetByName(name); if (!s) { s = ss.insertSheet(name); if (headers && headers.length) { s.appendRow(headers); s.getRange('1:1').setFontWeight('bold').setBackground('#2C3E35').setFontColor('#fff'); s.setFrozenRows(1); } } return s; } function doGet(e) { try { const name = (e.parameter || {}).sheet || ''; // ping 測試：無 sheet 參數時直接回傳連線成功 if (!name) return json({ status: 'ok', msg: '連線成功（通用版 GAS）' }); const ss = SpreadsheetApp.getActiveSpreadsheet(); const s = ss.getSheetByName(name); if (!s) return json({ status: 'ok', rows: [] }); const values = s.getDataRange().getValues(); const headers = values[0] || []; const rows = values.slice(1).map(r => { const o = {}; headers.forEach((h, i) => o[h] = r[i]); return o; }); return json({ status: 'ok', headers, rows }); } catch (err) { return json({ status: 'error', msg: err.message }); } } function doPost(e) { try { const d = JSON.parse(e.postData.contents); const s = getOrCreate(d.sheet, d.headers || []); if (d.append) { // 新增一列模式（銷售記錄） const headers = s.getRange(1, 1, 1, Math.max(s.getLastColumn(), 1)).getValues()[0]; // 自動新增尚未存在的商品欄位 if (d.ensureCols) { d.ensureCols.forEach(col => { if (!headers.includes(col)) { const idx = headers.indexOf(d.insertBefore || '結帳金額'); const at = (idx >= 0 ? idx : headers.length) + 1; s.insertColumnBefore(at); s.getRange(1, at).setValue(col) .setFontWeight('bold').setBackground('#2C3E35').setFontColor('#fff'); headers.splice(at - 1, 0, col); } }); } s.appendRow(headers.map(h => d.row[h] !== undefined ? d.row[h] : '')); return json({ status: 'ok' }); } // 覆寫整頁模式（商品 / 人員 / 院區） s.clearContents(); if (d.headers && d.headers.length) { s.appendRow(d.headers); s.getRange('1:1').setFontWeight('bold').setBackground('#2C3E35').setFontColor('#fff'); s.setFrozenRows(1); } (d.rows || []).forEach(row => { s.appendRow((d.headers || []).map(h => row[h] !== undefined ? row[h] : '')); }); return json({ status: 'ok', count: (d.rows || []).length }); } catch (err) { return json({ status: 'error', msg: err.message }); } } function json(o) { return ContentService.createTextOutput(JSON.stringify(o)).setMimeType(ContentService.MimeType.JSON); }
